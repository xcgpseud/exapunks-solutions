LINK 800

; X TRACKS OUR ROW
COPY 3 X

; T TRACKS OUR COLUMN
COPY 9 T

MARK LOOP_ROW
SUBI 3 X #DATA
MARK LOOP_COL
SUBI 9 T #DATA
MARK LOOP_VAL
COPY M #DATA
SUBI T 1 T

; WHEN 0 THIS WILL NOT
; GO TO LOOP_ROW
TJMP LOOP_ROW

; NEXT ROW
SUBI X 1 X

; END IF WE MUST
; WE CAN USE T FOR THIS
; AS IT WILL BE 9 AGAIN
TEST X = 0
TJMP END

; BACK TO FIRST COLUMN
COPY 9 T

; GOGO!
JUMP LOOP_ROW

; EMPTY END BLOCK
MARK END
