REPL GO
GRAB 300
COPY F M
DROP

MARK GO
LINK 800

GRAB 199
COPY M X

; MARGINALLY BETTER
; IF WE START AT THE END
SEEK 9999
SEEK -3

; SEARCHING THIS FILE
; FOR THE FILE ID
; OF OUR USERNAME
MARK FIND_FID
TEST X = F
TJMP SEND_FID

SEEK -4
JUMP FIND_FID

; SEND THE EXA TO THE
; FILE
MARK SEND_FID
SEEK 1
COPY F X
DROP

LINK 799

GRAB X

SEEK 2

COPY 0 X

; SUM UP ALL OF THE NUMS
; ALSO OVERWRITE WITH
; 75 AS WE GO TO SAVE
; LOOPING AGAIN. THIS
; MARGINALLY IMPROVES
; CYCLES.
MARK SUM
ADDI X F X
SEEK -1
COPY 75 F
SUBI X 75 X
TEST EOF
FJMP SUM

MARK OVERWRITE
COPY 75 F
SUBI X 75 X

TEST X < 75
FJMP OVERWRITE

COPY X F
